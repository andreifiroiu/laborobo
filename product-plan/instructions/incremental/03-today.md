import { TodayView } from '@/sections/today/components'

    tasks,

      blockers={blockers}
      upcomingDeadlines={upcomingDeadlines}
      activities={activities}

      onApprove={(id) => console.log('Approve:', id)}
      onReject={(id) => console.log('Reject:', id)}
      // ... other callbacks
    />
  )
}
